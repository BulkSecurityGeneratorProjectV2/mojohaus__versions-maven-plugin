48  /**
49   * Compare dependency versions of the current project to dependencies or dependency management of a remote repository
50   * project. Can optionally update locally the project instead of reporting the comparison
51   *
52   * @author Paul Gier
53   * @since 1.3
54   */
55  @Mojo( name = "compare-dependencies", requiresProject = true, requiresDirectInvocation = true, threadSafe = true )
56  public class CompareDependenciesMojo
57      extends AbstractVersionsDependencyUpdaterMojo
58  {
59  
60      /**
61       * The width to pad info messages.
62       *
63       * @since 1.0-alpha-1
64       */
65      private static final int INFO_PAD_SIZE = 68;
66  
67      /**
68       * The groupId, artifactId, and version of the remote project (POM) to which we are comparing. This should be in the
69       * form "groupId:artifactId:version"
70       */
71      @Parameter( property = "remotePom", required = true )
72      protected String remotePom;
73  
74      /**
75       * Ignore the list of remote dependencies and only compare the remote dependencyManagement
76       */
77      @Parameter( property = "ignoreRemoteDependencies", defaultValue = "false" )
78      protected boolean ignoreRemoteDependencies;
79  
80      /**
81       * Ignore the remote dependency management and only check against the actual dependencies of the remote project
82       */
83      @Parameter( property = "ignoreRemoteDependencyManagement", defaultValue = "false" )
84      protected boolean ignoreRemoteDependencyManagement;
85  
86      /**
87       * Update dependency versions in the current POM.
88       */
89      @Parameter( property = "updateDependencies", defaultValue = "false" )
90      protected boolean updateDependencies;
91  
92      /**
93       * Update dependency versions stored in properties
94       */
95      @Parameter( property = "updatePropertyVersions", defaultValue = "false" )
96      protected boolean updatePropertyVersions;
97  
98      /**
99       * Display the dependency version differences on the command line, but do not update the versions in the current
100      * pom. If updateDependencies is set to "true" this will automatically be set to false.
101      */
102     @Parameter( property = "reportMode", defaultValue = "true" )
103     protected boolean reportMode;
104 
105     /**
106      * If the output file is set, the diff report will be written to this file.
107      */
108     @Parameter( property = "reportOutputFile" )
109     protected File reportOutputFile;
110 
111     /**
112      * The project builder used to initialize the remote project.
113      */
114     @Component
115     protected MavenProjectBuilder mavenProjectBuilder;

119     /**
120      * @param pom the pom to update.
121      * @throws org.apache.maven.plugin.MojoExecutionException Something wrong with the plugin itself
122      * @throws org.apache.maven.plugin.MojoFailureException   The plugin detected an error in the build
123      * @throws javax.xml.stream.XMLStreamException            when things go wrong with XML streaming
124      * @see AbstractVersionsUpdaterMojo#update(org.codehaus.mojo.versions.rewriting.ModifiedPomXMLEventReader)
125      */
126     protected void update( ModifiedPomXMLEventReader pom )
127         throws MojoExecutionException, MojoFailureException, XMLStreamException
128     {
129         if ( this.ignoreRemoteDependencies && this.ignoreRemoteDependencyManagement )
130         {
131             throw new MojoFailureException( " ignoreRemoteDependencies and ignoreRemoteDependencyManagement"
132                                                 + "are both set to true.  At least one of these needs to be false " );
133         }
134 
135         if ( updateDependencies )
136         {
137             reportMode = false;
138         }
139 
140         String[] remotePomParts = this.remotePom.split( ":" );
141         if ( remotePomParts.length != 3 )
142         {
143             throw new MojoFailureException( " Invalid format for remotePom: " + remotePom );
144         }
145         String rGroupId = remotePomParts[0];
146         String rArtifactId = remotePomParts[1];
147         String rVersion = remotePomParts[2];
148 
149         Dependency remoteDependency = new Dependency();
150         remoteDependency.setGroupId( rGroupId );
151         remoteDependency.setArtifactId( rArtifactId );
152         remoteDependency.setVersion( rVersion );
153 
154         Artifact remoteArtifact = this.toArtifact( remoteDependency );
155         MavenProject remoteMavenProject = null;
156         try
157         {
158             remoteMavenProject =
159                 mavenProjectBuilder.buildFromRepository( remoteArtifact, remoteArtifactRepositories, localRepository );
160         }
161         catch ( ProjectBuildingException e )
162         {
163             throw new MojoExecutionException( "Unable to build remote project " + remoteArtifact, e );
164         }
165 
166         Map<String, Dependency> remoteDepsMap = new HashMap<String, Dependency>();
167         if ( !ignoreRemoteDependencyManagement )
168         {
169             List<Dependency> remoteProjectDepMgmtDeps = ( remoteMavenProject.getDependencyManagement() == null ) ? null
170                 : remoteMavenProject.getDependencyManagement().getDependencies();
171             mapDependencies( remoteDepsMap, remoteProjectDepMgmtDeps );
172         }
173         if ( !ignoreRemoteDependencies )
174         {
175             List<Dependency> remoteProjectDeps = remoteMavenProject.getDependencies();
176             mapDependencies( remoteDepsMap, remoteProjectDeps );
177         }
178 
179         List<String> totalDiffs = new ArrayList<>();
180         List<String> propertyDiffs = new ArrayList<>();
181         if ( getProject().getDependencyManagement() != null && isProcessingDependencyManagement() )
182         {
183             List<String> depManDiffs =
184                 compareVersions( pom, getProject().getDependencyManagement().getDependencies(), remoteDepsMap );
185             totalDiffs.addAll( depManDiffs );
186         }
187         if ( getProject().getDependencies() != null && isProcessingDependencies() )
188         {
189             List<String> depDiffs = compareVersions( pom, getProject().getDependencies(), remoteDepsMap );
190             totalDiffs.addAll( depDiffs );
191         }
192         if ( updatePropertyVersions )
193         {
194             Map<Property, PropertyVersions> versionProperties =
195                 this.getHelper().getVersionPropertiesMap( getProject(), null, null, null, true );
196             List<String> diff = updatePropertyVersions( pom, versionProperties, remoteDepsMap );
197             propertyDiffs.addAll( diff );
198         }
199 
200         if ( reportMode )
201         {
202             getLog().info( "The following differences were found:" );
203             if ( totalDiffs.size() == 0 )
204             {
205                 getLog().info( "  none" );
206             }
207             else
208             {
209                 for ( String totalDiff : totalDiffs )
210                 {
211                     getLog().info( "  " + totalDiff );
212                 }
213             }
214             getLog().info( "The following property differences were found:" );
215             if ( propertyDiffs.size() == 0 )
216             {
217                 getLog().info( "  none" );
218             }
219             else
220             {
221                 for ( String propertyDiff : propertyDiffs )
222                 {
223                     getLog().info( "  " + propertyDiff );
224                 }
225             }
226         }
227 
228         if ( reportOutputFile != null )
229         {
230             writeReportFile( totalDiffs, propertyDiffs );
231         }
232 
233     }

235     /**
236      * Compare the dependency versions of the current project with the dependency versions of a remote project
237      *
238      * @throws XMLStreamException
239      */
240     private List<String> compareVersions( ModifiedPomXMLEventReader pom, List<Dependency> dependencies,
241                                           Map<String, Dependency> remoteDependencies )
242         throws MojoExecutionException, XMLStreamException
243     {
244         List<String> updates = new ArrayList<>();
245         for ( Dependency dep : dependencies )
246         {
247             Artifact artifact = this.toArtifact( dep );
248             if ( !isIncluded( artifact ) )
249             {
250                 continue;
251             }
252 
253             Dependency remoteDep = remoteDependencies.get( dep.getManagementKey() );
254             if ( remoteDep != null )
255             {
256                 String remoteVersion = remoteDep.getVersion();
257 
258                 if ( !dep.getVersion().equals( remoteVersion ) )
259                 {
260                     StringBuilder buf = writeDependencyDiffMessage( dep, remoteVersion );
261                     updates.add( buf.toString() );
262                     if ( !reportMode )
263                     {
264                         if ( PomHelper.setDependencyVersion( pom, dep.getGroupId(), dep.getArtifactId(),
265                                                              dep.getVersion(), remoteVersion,
266                                                              getProject().getModel() ) )
267                         {
268                             getLog().info( "Updated " + toString( dep ) + " to version " + remoteVersion );
269                         }
270                     }
271 
272                 }
273 
274             }
275         }
276 
277         return updates;
278 
279     }

281     /**
282      * Updates the properties holding a version if necessary.
283      */
284     private List<String> updatePropertyVersions( ModifiedPomXMLEventReader pom,
285                                                  Map<Property, PropertyVersions> versionProperties,
286                                                  Map<String, Dependency> remoteDependencies )
287         throws XMLStreamException
288     {
289         List<String> result = new ArrayList<>();
290         for ( Map.Entry<Property, PropertyVersions> entry : versionProperties.entrySet() )
291         {
292             Property property = entry.getKey();
293             PropertyVersions version = entry.getValue();
294 
295             String candidateVersion = computeCandidateVersion( remoteDependencies, property, version );
296             if ( candidateVersion != null )
297             {
298                 String originalVersion = version.getAssociations()[0].getArtifact().getVersion(); // Yekes
299                 if ( !candidateVersion.equals( originalVersion ) ) // Update needed
300                 {
301                     result.add( writeDiffMessage( property.getName(), originalVersion, candidateVersion ).toString() );
302                     if ( !reportMode
303                         && PomHelper.setPropertyVersion( pom, null, property.getName(), candidateVersion ) )
304                     {
305                         getLog().info( "Updated ${" + property.getName() + "} from " + originalVersion + " to "
306                                            + candidateVersion );
307                     }
308                 }
309             }
310         }
311         return result;
312     }

314     /**
315      * Returns the candidate version to use for the specified property.
316      * <p/>
317      * The dependencies currently linked to the property must all be defined by the remote POM and they should refer to
318      * the same version. If that's the case, that same version is returned. Otherwise, <tt>null</tt> is returned
319      * indicating that there is no candidate.
320      *
321      * @param remoteDependencies the remote dependencies
322      * @param property           the property to update
323      * @param propertyVersions   the association
324      * @return the candidate version or <tt>null</tt> if there isn't any
325      */
326     private String computeCandidateVersion( Map<String, Dependency> remoteDependencies, Property property,
327                                             PropertyVersions propertyVersions )
328     {
329         String candidateVersion = null;
330         for ( ArtifactAssociation artifactAssociation : propertyVersions.getAssociations() )
331         {
332             String id = generateId( artifactAssociation.getArtifact() );
333             Dependency dependency = remoteDependencies.get( id );
334             if ( dependency == null )
335             {
336                 getLog().info( "Not updating ${" + property.getName() + "}: no info for " + id );
337                 return null;
338             }
339             else
340             {
341                 if ( candidateVersion == null )
342                 {
343                     candidateVersion = dependency.getVersion();
344                 }
345                 else if ( !candidateVersion.equals( dependency.getVersion() ) )
346                 {
347                     getLog().warn( "Could not update ${" + property.getName() + "}: version mismatch" );
348                     return null;
349                 }
350             }
351         }
352         return candidateVersion;
353     }

355     private void writeReportFile( List<String> dependenciesUpdate, List<String> propertiesUpdate )
356         throws MojoExecutionException
357     {
358         if ( !reportOutputFile.getParentFile().exists() )
359         {
360             reportOutputFile.getParentFile().mkdirs();
361         }
362 
363         try ( FileWriter fw = new FileWriter( reportOutputFile ); //
364               PrintWriter pw = new PrintWriter( fw ) )
365         {
366             pw.println( "The following differences were found:" );
367             pw.println();
368             if ( dependenciesUpdate.size() == 0 )
369             {
370                 pw.println( "  none" );
371             }
372             else
373             {
374                 for ( String dependencyUpdate : dependenciesUpdate )
375                 {
376                     pw.println( "  " + dependencyUpdate );
377                 }
378             }
379             pw.println();
380             pw.println( "The following property differences were found:" );
381             pw.println();
382             if ( propertiesUpdate.size() == 0 )
383             {
384                 pw.println( "  none" );
385             }
386             else
387             {
388                 for ( String propertyUpdate : propertiesUpdate )
389                 {
390                     pw.println( "  " + propertyUpdate );
391                 }
392             }
393         }
394         catch ( IOException e )
395         {
396             throw new MojoExecutionException( "Unable to write report file. ", e );
397         }
398     }

400     /**
401      * Create a simple message describing the version diff
402      *
403      * @param dep
404      * @param remoteVersion
405      * @return The message
406      */
407     private StringBuilder writeDependencyDiffMessage( Dependency dep, String remoteVersion )
408     {
409         String id = dep.getGroupId() + ":" + dep.getArtifactId();
410         return writeDiffMessage( id, dep.getVersion(), remoteVersion );
411     }
412 
413     private StringBuilder writeDiffMessage( String id, String originalVersion, String targetVersion )
414     {
415         StringBuilder buf = new StringBuilder();
416         buf.append( id );
417         buf.append( ' ' );
418         int padding = INFO_PAD_SIZE - originalVersion.length() - targetVersion.length() - 4;
419         while ( buf.length() < padding )
420         {
421             buf.append( '.' );
422         }
423         buf.append( ' ' );
424         buf.append( originalVersion );
425         buf.append( " -> " );
426         buf.append( targetVersion );
427         return buf;
428     }

430     /**
431      * Add a list of dependencies to a Map for easy access
432      *
433      * @param map
434      * @param deps
435      */
436     private void mapDependencies( Map<String, Dependency> map, List<Dependency> deps )
437     {
438         if ( deps != null )
439         {
440             for ( Dependency nextDep : deps )
441             {
442                 map.put( nextDep.getManagementKey(), nextDep );
443             }
444         }
445     }

447     /**
448      * Creates a key that is similar to what {@link Dependency#getManagementKey()} generates for a dependency.
449      */
450     private static String generateId( Artifact artifact )
451     {
452         StringBuilder sb = new StringBuilder();
453         sb.append( artifact.getGroupId() ).append( ":" ).append( artifact.getArtifactId() ).append( ":" )
454             .append( artifact.getType() );
455         if ( artifact.getClassifier() != null )
456         {
457             sb.append( ":" ).append( artifact.getClassifier() );
458         }
459         return sb.toString();
460     }
