3   import java.util.Map;
4   import java.util.function.Function;
5   import java.util.stream.Collectors;
6   
7   /**
8    * Miscellaneous utility class.
9    */
10  public class MiscUtils
11  {
12      /**
13       * Filters a given map leaving only elements fulfilling a predicate. Does not change the input map,
14       * the filtered map is returned as output.
15       *
16       * @param map       input map to be filtered
17       * @param predicate predicate for element comparison
18       * @param <K>       key type
19       * @param <V>       value type
20       * @return map such that every element comforms with the predicate
21       */
22      public static <K, V> Map<K, V> filter( Map<K, V> map,
23                                             Function<V, Boolean> predicate )
24      {
25          return map.entrySet().stream().filter( e -> predicate.apply( e.getValue() ) )
26                  .collect( Collectors.toMap( Map.Entry::getKey, Map.Entry::getValue ) );
27      }
28  }
