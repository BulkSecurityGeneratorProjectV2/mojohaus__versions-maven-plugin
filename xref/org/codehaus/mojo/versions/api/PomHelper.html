153     /**
154      * Searches the pom re-defining the specified property to the specified version.
155      *
156      * @param pom       The pom to modify.
157      * @param profileId The profile in which to modify the property.
158      * @param property  The property to modify.
159      * @param value     The new value of the property.
160      * @return <code>true</code> if a replacement was made.
161      * @throws XMLStreamException if somethinh went wrong.
162      */
163     public static boolean setPropertyVersion( final ModifiedPomXMLEventReader pom, final String profileId,
164                                               final String property, final String value )
165         throws XMLStreamException
166     {
167         Stack<String> stack = new Stack<>();
168         String path = "";
169         final Pattern propertyRegex;
170         final Pattern matchScopeRegex;
171         final Pattern projectProfileId;
172         boolean inMatchScope = false;
173         boolean madeReplacement = false;
174         if ( profileId == null )
175         {
176             propertyRegex = Pattern.compile( "/project/properties/" + RegexUtils.quote( property ) );
177             matchScopeRegex = Pattern.compile( "/project/properties" );
178             projectProfileId = null;
179         }
180         else
181         {
182             propertyRegex = Pattern.compile( "/project/profiles/profile/properties/" + RegexUtils.quote( property ) );
183             matchScopeRegex = Pattern.compile( "/project/profiles/profile" );
184             projectProfileId = Pattern.compile( "/project/profiles/profile/id" );
185         }
186 
187         pom.rewind();
188 
189         while ( pom.hasNext() )
190         {
191             XMLEvent event = pom.nextEvent();
192             if ( event.isStartElement() )
193             {
194                 stack.push( path );
195                 path = path + "/" + event.asStartElement().getName().getLocalPart();
196 
197                 if ( propertyRegex.matcher( path ).matches() )
198                 {
199                     pom.mark( 0 );
200                 }
201                 else if ( matchScopeRegex.matcher( path ).matches() )
202                 {
203                     // we're in a new match scope
204                     // reset any previous partial matches
205                     inMatchScope = profileId == null;
206                     pom.clearMark( 0 );
207                     pom.clearMark( 1 );
208                 }
209                 else if ( profileId != null && projectProfileId.matcher( path ).matches() )
210                 {
211                     String candidateId = pom.getElementText();
212                     path = stack.pop(); // since getElementText will be after the end element
213 
214                     inMatchScope = profileId.trim().equals( candidateId.trim() );
215                 }
216             }
217             if ( event.isEndElement() )
218             {
219                 if ( propertyRegex.matcher( path ).matches() )
220                 {
221                     pom.mark( 1 );
222                 }
223                 else if ( matchScopeRegex.matcher( path ).matches() )
224                 {
225                     if ( inMatchScope && pom.hasMark( 0 ) && pom.hasMark( 1 ) )
226                     {
227                         pom.replaceBetween( 0, 1, value );
228                         madeReplacement = true;
229                     }
230                     pom.clearMark( 0 );
231                     pom.clearMark( 1 );
232                     inMatchScope = false;
233                 }
234                 path = stack.pop();
235             }
236         }
237         return madeReplacement;
238     }

274     /**
275      * Sets the value of the given element given its parent element path.
276      * Will only consider the first found occurrence of the parent element.
277      * If the element is not found in the parent element, the method will create the element
278      * if {@code shouldCreate} is {@code true}.
279      *
280      * @param pom           pom to modify
281      * @param parentPath    path of the parent element
282      * @param elementName   name of the element to set or create
283      * @param value         the new value of the element
284      * @param shouldCreate  should the element be created if it's not found in the first encountered parent element
285      *                      matching the parentPath
286      * @return              {@code true} if the element was created or replaced
287      * @throws XMLStreamException if something went wrong
288      */
289     public static boolean setElementValue( ModifiedPomXMLEventReader pom, String parentPath,
290                                            String elementName, String value, boolean shouldCreate )
291             throws XMLStreamException
292     {
293         class ElementValueInternal
294         {
295             private final String parentName;
296             private final String superParentPath;
297 
298             private static final int MARK_CHILD_BEGIN = 0;
299             private static final int MARK_OPTION = 1;
300             private static final int PARENT_BEGIN = 2;
301 
302             ElementValueInternal()
303             {
304                 int lastDelimeterIndex = parentPath.lastIndexOf( '/' );
305                 parentName = parentPath.substring( lastDelimeterIndex + 1 );
306                 superParentPath = parentPath.substring( 0, lastDelimeterIndex );
307             }
308 
309             boolean process( String currentPath ) throws XMLStreamException
310             {
311                 boolean replacementMade = false;
312                 while ( !replacementMade && pom.hasNext() )
313                 {
314                     XMLEvent event = pom.nextEvent();
315                     if ( event.isStartElement() )
316                     {
317                         String currentElementName = event.asStartElement().getName().getLocalPart();
318 
319                         // here, we will only mark the beginning of the child or the parent element
320                         if ( currentPath.equals( parentPath ) && elementName.equals( currentElementName ) )
321                         {
322                             pom.mark( MARK_CHILD_BEGIN );
323                         }
324                         else if ( currentPath.equals( superParentPath ) && currentElementName.equals( parentName ) )
325                         {
326                               pom.mark( PARENT_BEGIN );
327                         }
328                         // process child element
329                         replacementMade = process( currentPath + "/" + currentElementName );
330                     }
331                     else if ( event.isEndElement() )
332                     {
333                         // here we're doing the replacement
334                         if ( currentPath.equals( parentPath + "/" + elementName ) )
335                         {
336                             // end of the child
337                             replaceValueInChild();
338                             replacementMade = true;
339                         }
340                         else if ( shouldCreate && currentPath.equals( parentPath ) )
341                         {
342                             // end of the parent
343                             replaceValueInParent();
344                             replacementMade = true;
345                         }
346                         else
347                         {
348                             return false;
349                         }
350                     }
351                 }
352                 return replacementMade;
353             }
354 
355             private void replaceValueInChild()
356             {
357                 pom.mark( MARK_OPTION );
358                 if ( pom.getBetween( MARK_CHILD_BEGIN, MARK_OPTION ).length() > 0 )
359                 {
360                     pom.replaceBetween( 0, 1, value );
361                 }
362                 else
363                 {
364                     pom.replace( String.format( "<%1$s>%2$s</%1$s>", elementName, value ) );
365                 }
366             }
367 
368             private void replaceValueInParent()
369             {
370                 pom.mark( MARK_OPTION );
371                 if ( pom.hasMark( PARENT_BEGIN ) )
372                 {
373                     if ( pom.getBetween( PARENT_BEGIN, MARK_OPTION ).length() > 0 )
374                     {
375                         pom.replace( String.format( "<%2$s>%3$s</%2$s></%1$s>",
376                                 parentName, elementName, value ) );
377                     }
378                     else
379                     {
380                         pom.replace( String.format( "<%1$s><%2$s>%3$s</%2$s></%1$s>",
381                                 parentName, elementName, value ) );
382                     }
383                 }
384                 else
385                 {
386                     pom.replace( String.format( "<%1$s><%2$s>%3$s</%2$s></%1$s>",
387                             parentName, elementName, value ) );
388                 }
389             }
390         }
391 
392         try
393         {
394             pom.rewind();
395             return new ElementValueInternal().process( "" );
396         }
397         finally
398         {
399             range( 0, 3 ).forEach( pom::clearMark );
400         }
401     }

417

419

441                     }
442                     pom.clearMark( 0 );
443                     pom.clearMark( 1 );
444                 }
445                 path = stack.pop();
446             }
447         }
448         return null;

563     /**
564      * Searches the pom re-defining the specified dependency to the specified version.
565      *
566      * @param pom        The pom to modify.
567      * @param groupId    The groupId of the dependency.
568      * @param artifactId The artifactId of the dependency.
569      * @param oldVersion The old version of the dependency.
570      * @param newVersion The new version of the dependency.
571      * @param model      The model to get the project properties from.
572      * @return <code>true</code> if a replacement was made.
573      * @throws XMLStreamException if something went wrong.
574      */
575     @SuppressWarnings( "checkstyle:MethodLength" )
576     public static boolean setDependencyVersion( final ModifiedPomXMLEventReader pom, final String groupId,
577                                                 final String artifactId, final String oldVersion,
578                                                 final String newVersion, final Model model )
579         throws XMLStreamException
580     {
581         Stack<String> stack = new Stack<>();
582         String path = "";
583 
584         Set<String> implicitPaths =
585             new HashSet<>( Arrays.asList( "/project/parent/groupId", "/project/parent/artifactId",
586                                           "/project/parent/version", "/project/groupId",
587                                           "/project/artifactId", "/project/version" ) );
588         Map<String, String> implicitProperties = new HashMap<>();
589 
590         for ( Map.Entry<Object, Object> entry : model.getProperties().entrySet() )
591         {
592             implicitProperties.put( (String) entry.getKey(), (String) entry.getValue() );
593         }
594 
595         pom.rewind();
596 
597         while ( pom.hasNext() )
598         {
599             while ( pom.hasNext() )
600             {
601                 XMLEvent event = pom.nextEvent();
602                 if ( event.isStartElement() )
603                 {
604                     stack.push( path );
605                     final String elementName = event.asStartElement().getName().getLocalPart();
606                     path = path + "/" + elementName;
607 
608                     if ( implicitPaths.contains( path ) )
609                     {
610                         final String elementText = pom.getElementText().trim();
611                         implicitProperties.put( path.substring( 1 ).replace( '/', '.' ), elementText );
612                         path = stack.pop();
613                     }
614                 }
615                 if ( event.isEndElement() )
616                 {
617                     path = stack.pop();
618                 }
619             }
620         }
621 
622         boolean modified = true;
623         while ( modified )
624         {
625             modified = false;
626             for ( Map.Entry<String, String> entry : implicitProperties.entrySet() )
627             {
628                 if ( entry.getKey().contains( ".parent" ) )
629                 {
630                     String child = entry.getKey().replace( ".parent", "" );
631                     if ( !implicitProperties.containsKey( child ) )
632                     {
633                         implicitProperties.put( child, entry.getValue() );
634                         modified = true;
635                         break;
636                     }
637                 }
638             }
639         }
640 
641         stack = new Stack<>();
642         path = "";
643         boolean inMatchScope = false;
644         boolean madeReplacement = false;
645         boolean haveGroupId = false;
646         boolean haveArtifactId = false;
647         boolean haveOldVersion = false;
648 
649         final Pattern matchScopeRegex = Pattern.compile(
650             "/project" + "(/profiles/profile)?"
651                 + "((/dependencyManagement)|(/build(/pluginManagement)?/plugins/plugin))?"
652                 + "/dependencies/dependency" );
653 
654         final Pattern matchTargetRegex = Pattern.compile(
655             "/project" + "(/profiles/profile)?"
656                 + "((/dependencyManagement)|(/build(/pluginManagement)?/plugins/plugin))?"
657                 + "/dependencies/dependency"
658                 + "((/groupId)|(/artifactId)|(/version))" );
659 
660         pom.rewind();
661 
662         while ( pom.hasNext() )
663         {
664             XMLEvent event = pom.nextEvent();
665             if ( event.isStartElement() )
666             {
667                 stack.push( path );
668                 final String elementName = event.asStartElement().getName().getLocalPart();
669                 path = path + "/" + elementName;
670 
671                 if ( matchScopeRegex.matcher( path ).matches() )
672                 {
673                     // we're in a new match scope
674                     // reset any previous partial matches
675                     inMatchScope = true;
676                     pom.clearMark( 0 );
677                     pom.clearMark( 1 );
678 
679                     haveGroupId = false;
680                     haveArtifactId = false;
681                     haveOldVersion = false;
682                 }
683                 else if ( inMatchScope && matchTargetRegex.matcher( path ).matches() )
684                 {
685                     if ( "groupId".equals( elementName ) )
686                     {
687                         haveGroupId = groupId.equals( evaluate( pom.getElementText().trim(), implicitProperties ) );
688                         path = stack.pop();
689                     }
690                     else if ( "artifactId".equals( elementName ) )
691                     {
692                         haveArtifactId =
693                             artifactId.equals( evaluate( pom.getElementText().trim(), implicitProperties ) );
694                         path = stack.pop();
695                     }
696                     else if ( "version".equals( elementName ) )
697                     {
698                         pom.mark( 0 );
699                     }
700                 }
701             }
702             if ( event.isEndElement() )
703             {
704                 if ( matchTargetRegex.matcher( path ).matches()
705                     && "version".equals( event.asEndElement().getName().getLocalPart() ) )
706                 {
707                     pom.mark( 1 );
708                     String compressedPomVersion = StringUtils.deleteWhitespace( pom.getBetween( 0, 1 ).trim() );
709                     String compressedOldVersion = StringUtils.deleteWhitespace( oldVersion );
710 
711                     try
712                     {
713                         haveOldVersion = isVersionOverlap( compressedOldVersion, compressedPomVersion );
714                     }
715                     catch ( InvalidVersionSpecificationException e )
716                     {
717                         // fall back to string comparison
718                         haveOldVersion = compressedOldVersion.equals( compressedPomVersion );
719                     }
720                 }
721                 else if ( matchScopeRegex.matcher( path ).matches() )
722                 {
723                     if ( inMatchScope && pom.hasMark( 0 ) && pom.hasMark( 1 ) && haveGroupId && haveArtifactId
724                         && haveOldVersion )
725                     {
726                         pom.replaceBetween( 0, 1, newVersion );
727                         madeReplacement = true;
728                     }

742     /**
743      * A lightweight expression evaluation function.
744      *
745      * @param expr       The expression to evaluate.
746      * @param properties The properties to substitute.
747      * @return The evaluated expression.
748      */
749     public static String evaluate( String expr, Map<String, String> properties )
750     {
751         if ( expr == null )
752         {
753             return null;
754         }
755 
756         String expression = stripTokens( expr );
757         if ( expression.equals( expr ) )
758         {
759             int index = expr.indexOf( "${" );
760             if ( index >= 0 )
761             {
762                 int lastIndex = expr.indexOf( "}", index );
763                 if ( lastIndex >= 0 )
764                 {
765                     String retVal = expr.substring( 0, index );
766 
767                     if ( index > 0 && expr.charAt( index - 1 ) == '$' )
768                     {
769                         retVal += expr.substring( index + 1, lastIndex + 1 );
770                     }
771                     else
772                     {
773                         retVal += evaluate( expr.substring( index, lastIndex + 1 ), properties );
774                     }
775 
776                     retVal += evaluate( expr.substring( lastIndex + 1 ), properties );
777                     return retVal;
778                 }
779             }
780 
781             // Was not an expression
782             if ( expression.contains( "$$" ) )
783             {
784                 return expression.replaceAll( "\\$\\$", "\\$" );
785             }
786             else
787             {
788                 return expression;
789             }
790         }
791 
792         String value = properties.get( expression );
793 
794         if ( value != null )
795         {
796             int exprStartDelimiter = value.indexOf( "${" );
797 
798             if ( exprStartDelimiter >= 0 )
799             {
800                 if ( exprStartDelimiter > 0 )
801                 {
802                     value = value.substring( 0, exprStartDelimiter )
803                         + evaluate( value.substring( exprStartDelimiter ), properties );
804                 }
805                 else
806                 {
807                     value = evaluate( value.substring( exprStartDelimiter ), properties );
808                 }
809             }
810         }
811         else
812         {
813             // TODO find a way to log that and not use this System.out!!
814             // this class could be a component with logger injected !!
815             System.out.println( "expression: " + expression + " no value " );
816         }
817         return value == null ? expr : value;
818     }

836      * Checks if two versions or ranges have an overlap.
837      *
838      * @param leftVersionOrRange  the 1st version number or range to test
839      * @param rightVersionOrRange the 2nd version number or range to test
840      * @return true if both versions have an overlap
841      * @throws InvalidVersionSpecificationException if the versions can't be parsed to a range
842      */
843     public static boolean isVersionOverlap( String leftVersionOrRange, String rightVersionOrRange )
844         throws InvalidVersionSpecificationException
845     {
846         VersionRange pomVersionRange = createVersionRange( leftVersionOrRange );
847         if ( !pomVersionRange.hasRestrictions() )
848         {
849             return true;
850         }
851 
852         VersionRange oldVersionRange = createVersionRange( rightVersionOrRange );
853         if ( !oldVersionRange.hasRestrictions() )
854         {
855             return true;
856         }
857 
858         VersionRange result = oldVersionRange.restrict( pomVersionRange );
859         return result.hasRestrictions();
860     }

873     /**
874      * Searches the pom re-defining the specified plugin to the specified version.
875      *
876      * @param pom        The pom to modify.
877      * @param groupId    The groupId of the dependency.
878      * @param artifactId The artifactId of the dependency.
879      * @param oldVersion The old version of the dependency.
880      * @param newVersion The new version of the dependency.
881      * @return <code>true</code> if a replacement was made.
882      * @throws XMLStreamException if somethinh went wrong.
883      */
884     public static boolean setPluginVersion( final ModifiedPomXMLEventReader pom, final String groupId,
885                                             final String artifactId, final String oldVersion, final String newVersion )
886         throws XMLStreamException
887     {
888         Stack<String> stack = new Stack<>();
889         String path = "";
890         final Pattern matchScopeRegex;
891         final Pattern matchTargetRegex;
892         boolean inMatchScope = false;
893         boolean madeReplacement = false;
894         boolean haveGroupId = false;
895         boolean needGroupId = groupId != null && !APACHE_MAVEN_PLUGINS_GROUPID.equals( groupId );
896         boolean haveArtifactId = false;
897         boolean haveOldVersion = false;
898 
899         matchScopeRegex = Pattern.compile( "/project" + "(/profiles/profile)?"
900                                                + "((/build(/pluginManagement)?)|(/reporting))/plugins/plugin" );
901 
902         matchTargetRegex = Pattern.compile( "/project" + "(/profiles/profile)?"
903                                                 + "((/build(/pluginManagement)?)|(/reporting))/plugins/plugin"
904                                                 + "((/groupId)|(/artifactId)|(/version))" );
905 
906         pom.rewind();
907 
908         while ( pom.hasNext() )
909         {
910             XMLEvent event = pom.nextEvent();
911             if ( event.isStartElement() )
912             {
913                 stack.push( path );
914                 final String elementName = event.asStartElement().getName().getLocalPart();
915                 path = path + "/" + elementName;
916 
917                 if ( matchScopeRegex.matcher( path ).matches() )
918                 {
919                     // we're in a new match scope
920                     // reset any previous partial matches
921                     inMatchScope = true;
922                     pom.clearMark( 0 );
923                     pom.clearMark( 1 );
924 
925                     haveGroupId = false;
926                     haveArtifactId = false;
927                     haveOldVersion = false;
928                 }
929                 else if ( inMatchScope && matchTargetRegex.matcher( path ).matches() )
930                 {
931                     if ( "groupId".equals( elementName ) )
932                     {
933                         haveGroupId =  pom.getElementText().trim().equals( groupId );
934                         path = stack.pop();
935                     }
936                     else if ( "artifactId".equals( elementName ) )
937                     {
938                         haveArtifactId = artifactId.equals( pom.getElementText().trim() );
939                         path = stack.pop();
940                     }
941                     else if ( "version".equals( elementName ) )
942                     {
943                         pom.mark( 0 );
944                     }
945                 }
946             }
947             if ( event.isEndElement() )
948             {
949                 if ( matchTargetRegex.matcher( path ).matches()
950                     && "version".equals( event.asEndElement().getName().getLocalPart() ) )
951                 {
952                     pom.mark( 1 );
953 
954                     try
955                     {
956                         haveOldVersion = isVersionOverlap( oldVersion, pom.getBetween( 0, 1 ).trim() );
957                     }
958                     catch ( InvalidVersionSpecificationException e )
959                     {
960                         // fall back to string comparison
961                         haveOldVersion = oldVersion.equals( pom.getBetween( 0, 1 ).trim() );
962                     }
963                 }
964                 else if ( matchScopeRegex.matcher( path ).matches() )
965                 {
966                     if ( inMatchScope && pom.hasMark( 0 ) && pom.hasMark( 1 ) && ( haveGroupId || !needGroupId )
967                         && haveArtifactId && haveOldVersion )
968                     {
969                         pom.replaceBetween( 0, 1, newVersion );
970                         madeReplacement = true;
971                         pom.clearMark( 0 );
972                         pom.clearMark( 1 );
973                         haveArtifactId = false;
974                         haveGroupId = false;
975                         haveOldVersion = false;
976                     }
977                     inMatchScope = false;
978                 }
979                 path = stack.pop();
980             }
981         }
982         return madeReplacement;
983     }

986      * Traverses the project tree upwards, adding the raw models of every project it encounters to the map
987      *
988      * @param project maven project of the child for which the models need to be gathered
989      * @return gathered map of raw models per project
990      */
991     private static Map<MavenProject, Model> getRawModelWithParents( MavenProject project ) throws IOException
992     {
993         // constructs a tree sorted from children to parents
994         Map<MavenProject, Model> models = new TreeMap<>( ( p1, p2 ) ->
995         {
996             for ( MavenProject p = p1; p != null; p = p.getParent() )
997             {
998                 if ( p == p2 ) // meaning p2 is an ancestor to p1 or p1 == p2
999                 {
1000                     return p == p1
1001                             ? 0
1002                             : -1; // p1 is the child
1003                 }
1004             }
1005             return 1;
1006         } );
1007         for ( MavenProject p = project; p != null; p = p.getParent() )
1008         {
1009             models.put( p, p.getFile() != null
1010                     ? getRawModel( p )
1011                     : p.getOriginalModel() );
1012         }
1013         return models;

1018      *
1019      * @param helper  Our versions helper.
1020      * @param project The project to examine.
1021      * @return An array of properties that are associated within the project.
1022      * @throws ExpressionEvaluationException if an expression cannot be evaluated.
1023      * @throws IOException                   if the project's pom file cannot be parsed.
1024      * @since 1.0-alpha-3
1025      */
1026     public static PropertyVersionsBuilder[] getPropertyVersionsBuilders( VersionsHelper helper, MavenProject project )
1027             throws ExpressionEvaluationException, IOException
1028     {
1029         ExpressionEvaluator expressionEvaluator = helper.getExpressionEvaluator( project );
1030         Map<MavenProject, Model> reactorModels = getRawModelWithParents( project );
1031 
1032         Map<String, PropertyVersionsBuilder> propertiesMap = new TreeMap<>();
1033 
1034         Set<String> activeProfiles = new TreeSet<>();
1035         for ( Profile profile : project.getActiveProfiles() )
1036         {
1037             activeProfiles.add( profile.getId() );
1038         }
1039 
1040         // add any properties from profiles first (as they override properties from the project
1041         for ( Iterator<Profile> it = reactorModels.values().stream()
1042                 .flatMap( model -> model.getProfiles().stream() )
1043                 .filter( profile -> activeProfiles.contains( profile.getId() ) )
1044                 .iterator(); it.hasNext(); )
1045         {
1046             Profile profile = it.next();
1047             try
1048             {
1049                 addProperties( helper, propertiesMap, profile.getId(), profile.getProperties() );
1050                 if ( profile.getDependencyManagement() != null )
1051                 {
1052                     addDependencyAssocations( helper, expressionEvaluator, propertiesMap,
1053                             profile.getDependencyManagement().getDependencies(), false );
1054                 }
1055                 addDependencyAssocations( helper, expressionEvaluator, propertiesMap,
1056                         profile.getDependencies(),
1057                         false );
1058                 if ( profile.getBuild() != null )
1059                 {
1060                     if ( profile.getBuild().getPluginManagement() != null )
1061                     {
1062                         addPluginAssociations( helper, expressionEvaluator, propertiesMap,
1063                                 profile.getBuild().getPluginManagement().getPlugins() );
1064                     }
1065                     addPluginAssociations( helper, expressionEvaluator, propertiesMap,
1066                             profile.getBuild().getPlugins() );
1067                 }
1068                 if ( profile.getReporting() != null )
1069                 {
1070                     addReportPluginAssociations( helper, expressionEvaluator, propertiesMap,
1071                             profile.getReporting().getPlugins() );
1072                 }
1073             }
1074             catch ( ExpressionEvaluationException e )
1075             {
1076                 throw new RuntimeException( e );
1077             }
1078         }
1079 
1080         // second, we add all the properties in the pom
1081         reactorModels.values().forEach( model -> addProperties( helper, propertiesMap, null, model.getProperties() ) );
1082 
1083 
1084         for ( MavenProject currentPrj = project; currentPrj != null; currentPrj = currentPrj.getParent() )
1085         {
1086             Model model = reactorModels.get( currentPrj );
1087 
1088             if ( model.getDependencyManagement() != null )
1089             {
1090                 addDependencyAssocations( helper, expressionEvaluator, propertiesMap,
1091                         model.getDependencyManagement().getDependencies(), false );
1092             }
1093             addDependencyAssocations( helper, expressionEvaluator, propertiesMap, model.getDependencies(), false );
1094             if ( model.getBuild() != null )
1095             {
1096                 if ( model.getBuild().getPluginManagement() != null )
1097                 {
1098                     addPluginAssociations( helper, expressionEvaluator, propertiesMap,
1099                             model.getBuild().getPluginManagement().getPlugins() );
1100                 }
1101                 addPluginAssociations( helper, expressionEvaluator, propertiesMap, model.getBuild().getPlugins() );
1102             }
1103             if ( model.getReporting() != null )
1104             {
1105                 addReportPluginAssociations( helper, expressionEvaluator, propertiesMap,
1106                         model.getReporting().getPlugins() );
1107             }
1108 
1109             // third, we add any associations from the active profiles
1110             for ( Profile profile : model.getProfiles() )
1111             {
1112                 if ( !activeProfiles.contains( profile.getId() ) )
1113                 {
1114                     continue;
1115                 }
1116                 if ( profile.getDependencyManagement() != null )
1117                 {
1118                     addDependencyAssocations( helper, expressionEvaluator, propertiesMap,
1119                             profile.getDependencyManagement().getDependencies(), false );
1120                 }
1121                 addDependencyAssocations( helper, expressionEvaluator, propertiesMap, profile.getDependencies(),
1122                         false );
1123                 if ( profile.getBuild() != null )
1124                 {
1125                     if ( profile.getBuild().getPluginManagement() != null )
1126                     {
1127                         addPluginAssociations( helper, expressionEvaluator, propertiesMap,
1128                                 profile.getBuild().getPluginManagement().getPlugins() );
1129                     }
1130                     addPluginAssociations( helper, expressionEvaluator, propertiesMap,
1131                             profile.getBuild().getPlugins() );
1132                 }
1133                 if ( profile.getReporting() != null )
1134                 {
1135                     addReportPluginAssociations( helper, expressionEvaluator, propertiesMap,
1136                             profile.getReporting().getPlugins() );
1137                 }
1138             }
1139         }
1140 
1141         // finally, remove any properties without associations
1142         purgeProperties( propertiesMap );
1143 
1144         return propertiesMap.values().toArray( new PropertyVersionsBuilder[0] );
1145     }

1328     {

1331         Matcher m = lowerBound.matcher( rawVersionRange );
1332         if ( m.find() )
1333         {

1337             {
1338                 builder.addLowerBound( lowerLimit, includeLower );

1341         m = upperBound.matcher( rawVersionRange );
1342         if ( m.find() )
1343         {
1344             boolean includeUpper = "[".equals( m.group( 3 ) );
1345             String upperLimit = m.group( 2 );
1346             if ( StringUtils.isNotEmpty( upperLimit ) )
1347             {
1348                 builder.addUpperBound( upperLimit, includeUpper );

1535      *
1536      * @param project              The project to find the local root for.
1537      * @param builder              {@linkplain MavenProjectBuilder} object
1538      * @param localRepository      the local repo.
1539      * @param globalProfileManager the global profile manager.
1540      * @param logger               The logger to log tog
1541      * @return The local root (note this may be the project passed as an argument).
1542      */
1543     public static MavenProject getLocalRoot( MavenProjectBuilder builder, MavenProject project,
1544                                              ArtifactRepository localRepository, ProfileManager globalProfileManager,
1545                                              Log logger )
1546     {
1547         logger.info( "Searching for local aggregator root..." );
1548         while ( true )
1549         {
1550             final File parentDir = project.getBasedir().getParentFile();
1551             if ( parentDir != null && parentDir.isDirectory() )
1552             {
1553                 logger.debug( "Checking to see if " + parentDir + " is an aggregator parent" );
1554                 File parent = new File( parentDir, "pom.xml" );
1555                 if ( parent.isFile() )
1556                 {
1557                     try
1558                     {
1559                         final MavenProject parentProject =
1560                             builder.build( parent, localRepository, globalProfileManager );
1561                         if ( getAllChildModules( parentProject, logger ).contains( project.getBasedir().getName() ) )
1562                         {
1563                             logger.debug( parentDir + " is an aggregator parent" );
1564                             project = parentProject;
1565                             continue;
1566                         }
1567                         else
1568                         {
1569                             logger.debug( parentDir + " is not an aggregator parent" );
1570                         }
1571                     }
1572                     catch ( ProjectBuildingException e )
1573                     {
1574                         logger.warn( e );
1575                     }
1576                 }
1577             }
1578             logger.debug( "Local aggregation root is " + project.getBasedir() );
1579             return project;

1602     /**
1603      * Builds a sub-map of raw models keyed by module path.
1604      *
1605      * @param path    The relative path to base the sub-map on.
1606      * @param model   The model at the relative path.
1607      * @param project The project to build from.
1608      * @param logger  The logger for logging.
1609      * @return A map of raw models keyed by path relative to the project's basedir.
1610      * @throws IOException if things go wrong.
1611      */
1612     private static Map<String, Model> getReactorModels( String path, Model model, MavenProject project, Log logger )
1613         throws IOException
1614     {
1615         if ( path.length() > 0 && !path.endsWith( "/" ) )
1616         {
1617             path += '/';
1618         }
1619         Map<String, Model> result = new LinkedHashMap<>();
1620         Map<String, Model> childResults = new LinkedHashMap<>();
1621 
1622         File baseDir = path.length() > 0 ? new File( project.getBasedir(), path ) : project.getBasedir();
1623 
1624         Set<String> childModules = getAllChildModules( model, logger );
1625 
1626         removeMissingChildModules( logger, baseDir, childModules );
1627 
1628         for ( String moduleName : childModules )
1629         {
1630             String modulePath = path + moduleName;
1631 
1632             File moduleDir = new File( baseDir, moduleName );
1633 
1634             File moduleProjectFile;
1635 
1636             if ( moduleDir.isDirectory() )
1637             {
1638                 moduleProjectFile = new File( moduleDir, "pom.xml" );
1639             }
1640             else
1641             {
1642                 // i don't think this should ever happen... but just in case
1643                 // the module references the file-name
1644                 moduleProjectFile = moduleDir;
1645             }
1646 
1647             try
1648             {
1649                 // the aim of this goal is to fix problems when the project cannot be parsed by Maven
1650                 // so we have to work with the raw model and not the interpolated parsed model from maven
1651                 Model moduleModel = getRawModel( moduleProjectFile );
1652                 result.put( modulePath, moduleModel );
1653                 childResults.putAll( getReactorModels( modulePath, moduleModel, project, logger ) );
1654             }
1655             catch ( IOException e )
1656             {
1657                 logger.debug( "Could not parse " + moduleProjectFile.getPath(), e );
1658             }
1659         }
1660         result.putAll( childResults ); // more efficient update order if all children are added after siblings
1661         return result;
1662     }

1769     /**
1770      * Reads imported POMs from the dependency management section.
1771      *
1772      * @param pom POM
1773      * @return a non-null list of {@link Dependency} for each imported POM
1774      * @throws XMLStreamException XML stream exception
1775      * @see <a href="https://github.com/mojohaus/versions-maven-plugin/issues/134">bug #134</a>
1776      * @since 2.4
1777      */
1778     public static List<Dependency> readImportedPOMsFromDependencyManagementSection( ModifiedPomXMLEventReader pom )
1779         throws XMLStreamException
1780     {
1781         List<Dependency> importedPOMs = new ArrayList<>();
1782         Stack<String> stack = new Stack<>();
1783 
1784         String groupIdElement = "groupId";
1785         String artifactIdElement = "artifactId";
1786         String versionElement = "version";
1787         String typeElement = "type";
1788         String scopeElement = "scope";
1789         Set<String> recognizedElements =
1790             new HashSet<>( Arrays.asList( groupIdElement, artifactIdElement, versionElement, typeElement,
1791                                           scopeElement ) );
1792         Map<String, String> depData = new HashMap<>();
1793 
1794         pom.rewind();
1795 
1796         String depMgmtDependencyPath = "/project/dependencyManagement/dependencies/dependency";
1797 
1798         while ( pom.hasNext() )
1799         {
1800             XMLEvent event = pom.nextEvent();
1801 
1802             if ( event.isStartElement() )
1803             {
1804                 final String elementName = event.asStartElement().getName().getLocalPart();
1805                 String parent = "";
1806                 if ( !stack.isEmpty() )
1807                 {
1808                     parent = stack.peek();
1809                 }
1810                 String currentPath = parent + "/" + elementName;
1811                 stack.push( currentPath );
1812 
1813                 if ( currentPath.startsWith( depMgmtDependencyPath ) && recognizedElements.contains( elementName ) )
1814                 {
1815                     final String elementText = pom.getElementText().trim();
1816                     depData.put( elementName, elementText );
1817                     stack.pop();
1818                 }
1819             }
1820             if ( event.isEndElement() )
1821             {
1822                 String path = stack.pop();
1823                 if ( depMgmtDependencyPath.equals( path ) )
1824                 {
1825                     if ( "pom".equals( depData.get( typeElement ) ) && "import".equals( depData.get( scopeElement ) ) )
1826                     {
1827                         importedPOMs.add( DependencyBuilder.newBuilder()
1828                                 .withGroupId( depData.get( groupIdElement ) )
1829                                 .withArtifactId( depData.get( artifactIdElement ) )
1830                                 .withVersion( depData.get( versionElement ) )
1831                                 .withType( depData.get( typeElement ) )
1832                                 .withScope( depData.get( scopeElement ) )
1833                                 .build() );
1834                     }
1835                     depData.clear();
1836                 }
1837             }
1838         }
1839         return importedPOMs;
1840     }
