243     private synchronized void addChange( String groupId, String artifactId, String oldVersion, String newVersion )
244     {
245         if ( !newVersion.equals( oldVersion ) )
246         {
247             sourceChanges.add( new VersionChange( groupId, artifactId, oldVersion, newVersion ) );
248         }
249     }

251     /**
252      * Called when this mojo is executed.
253      *
254      * @throws org.apache.maven.plugin.MojoExecutionException when things go wrong.
255      * @throws org.apache.maven.plugin.MojoFailureException   when things go wrong.
256      */
257     public void execute() throws MojoExecutionException, MojoFailureException
258     {
259         if ( getProject().getOriginalModel().getVersion() == null )
260         {
261             throw new MojoExecutionException( "Project version is inherited from parent." );
262         }
263 
264         if ( removeSnapshot && !nextSnapshot )
265         {
266             String version = getVersion();
267             if ( version.endsWith( SNAPSHOT ) )
268             {
269                 String release = version.substring( 0, version.indexOf( SNAPSHOT ) );
270                 newVersion = release;
271                 getLog().info( "SNAPSHOT found.  BEFORE " + version + "  --> AFTER: " + newVersion );
272             }
273         }
274 
275         if ( !nextSnapshot && nextSnapshotIndexToIncrement != null )
276         {
277             throw new MojoExecutionException( "nextSnapshotIndexToIncrement is not valid when nextSnapshot is false" );
278         }
279 
280         if ( !removeSnapshot && nextSnapshot )
281         {
282             String version = getVersion();
283             newVersion = getIncrementedVersion( version, nextSnapshotIndexToIncrement );
284             getLog().info( "SNAPSHOT found.  BEFORE " + version + "  --> AFTER: " + newVersion );
285         }
286 
287         if ( StringUtils.isEmpty( newVersion ) )
288         {
289             if ( settings.isInteractiveMode() )
290             {
291                 try
292                 {
293                     newVersion = prompter.prompt( "Enter the new version to set",
294                                                   getProject().getOriginalModel().getVersion() );
295                 }
296                 catch ( PrompterException e )
297                 {
298                     throw new MojoExecutionException( e.getMessage(), e );
299                 }
300             }
301             else
302             {
303                 throw new MojoExecutionException( "You must specify the new version, either by using the newVersion "
304                                                       + "property (that is -DnewVersion=... on the command line) "
305                                                       + "or run in interactive mode" );
306             }
307         }
308         if ( StringUtils.isEmpty( newVersion ) )
309         {
310             throw new MojoExecutionException( "You must specify the new version, either by using the newVersion "
311                                                   + "property (that is -DnewVersion=... on the command line) "
312                                                   + "or run in interactive mode" );
313         }
314 
315         if ( !"onchange".equals( updateBuildOutputTimestampPolicy ) && !"always".equals(
316             updateBuildOutputTimestampPolicy ) && !"never".equals( updateBuildOutputTimestampPolicy ) )
317         {
318             throw new MojoExecutionException(
319                 "updateBuildOutputTimestampPolicy should be one of: " + "\"onchange\", \"always\", \"never\"." );
320         }
321 
322         try
323         {
324             final MavenProject project;
325             if ( processFromLocalAggregationRoot )
326             {
327                 project = PomHelper.getLocalRoot( projectBuilder, getProject(), localRepository, null, getLog() );
328             }
329             else
330             {
331                 project = getProject();
332             }
333 
334             getLog().info( "Local aggregation root: " + project.getBasedir() );
335             Map<String, Model> reactorModels = PomHelper.getReactorModels( project, getLog() );
336             final SortedMap<String, Model> reactor =
337                 new TreeMap<String, Model>( new ReactorDepthComparator( reactorModels ) );
338             reactor.putAll( reactorModels );
339 
340             // set of files to update
341             final Set<File> files = new LinkedHashSet<File>();
342 
343             getLog().info(
344                 "Processing change of " + groupId + ":" + artifactId + ":" + oldVersion + " -> " + newVersion );
345             Pattern groupIdRegex =
346                 Pattern.compile( RegexUtils.convertWildcardsToRegex( fixNullOrEmpty( groupId, "*" ), true ) );
347             Pattern artifactIdRegex =
348                 Pattern.compile( RegexUtils.convertWildcardsToRegex( fixNullOrEmpty( artifactId, "*" ), true ) );
349             Pattern oldVersionIdRegex =
350                 Pattern.compile( RegexUtils.convertWildcardsToRegex( fixNullOrEmpty( oldVersion, "*" ), true ) );
351             boolean found = false;
352             for ( Model m : reactor.values() )
353             {
354                 final String mGroupId = PomHelper.getGroupId( m );
355                 final String mArtifactId = PomHelper.getArtifactId( m );
356                 final String mVersion = PomHelper.getVersion( m );
357                 if ( ( processAllModules
358                         || groupIdRegex.matcher( mGroupId ).matches()
359                         && artifactIdRegex.matcher( mArtifactId ).matches() )
360                         && oldVersionIdRegex.matcher( mVersion ).matches()
361                         && !newVersion.equals( mVersion ) )
362                 {
363                     found = true;
364                     // if the change is not one we have swept up already
365                     applyChange( project, reactor, files, mGroupId, m.getArtifactId(),
366                             StringUtils.isBlank( oldVersion ) || "*".equals( oldVersion ) ? "" : mVersion );
367                 }
368             }
369             if ( !found && RegexUtils.getWildcardScore( groupId ) == 0 && RegexUtils.getWildcardScore( artifactId ) == 0
370                 && RegexUtils.getWildcardScore( oldVersion ) == 0 )
371             {
372                 applyChange( project, reactor, files, groupId, artifactId, oldVersion );
373             }
374 
375             // now process all the updates
376             for ( File file : files )
377             {
378                 process( file );
379             }
380 
381         }
382         catch ( IOException e )
383         {
384             throw new MojoExecutionException( e.getMessage(), e );
385         }
386     }

388     /**
389      * Returns the incremented version, with the nextSnapshotIndexToIncrement indicating the 1-based index,
390      * conunting from the left, or the most major version component, of the version string.
391      *
392      * @param version input version
393      * @return version with the incremented index specified by nextSnapshotIndexToIncrement or last index
394      * @throws MojoExecutionException thrown if the input parameters are invalid
395      */
396     protected String getIncrementedVersion( String version, Integer nextSnapshotIndexToIncrement )
397         throws MojoExecutionException
398     {
399         String versionWithoutSnapshot =
400             version.endsWith( SNAPSHOT ) ? version.substring( 0, version.indexOf( SNAPSHOT ) ) : version;
401         List<String> numbers = new LinkedList<>( Arrays.asList( versionWithoutSnapshot.split( "\\." ) ) );
402 
403         if ( nextSnapshotIndexToIncrement == null )
404         {
405             nextSnapshotIndexToIncrement = numbers.size();
406         }
407         else if ( nextSnapshotIndexToIncrement < 1 )
408         {
409             throw new MojoExecutionException( "nextSnapshotIndexToIncrement cannot be less than 1" );
410         }
411         else if ( nextSnapshotIndexToIncrement > numbers.size() )
412         {
413             throw new MojoExecutionException(
414                 "nextSnapshotIndexToIncrement cannot be greater than the last version index" );
415         }
416         int snapshotVersionToIncrement = Integer.parseInt( numbers.remove( nextSnapshotIndexToIncrement - 1 ) );
417         numbers.add( nextSnapshotIndexToIncrement - 1, String.valueOf( snapshotVersionToIncrement + 1 ) );
418 
419         return StringUtils.join( numbers.toArray( new String[0] ), "." ) + "-SNAPSHOT";
420     }

422     private static String fixNullOrEmpty( String value, String defaultValue )
423     {
424         return StringUtils.isBlank( value ) ? defaultValue : value;
425     }

427     private void applyChange( MavenProject project, SortedMap<String, Model> reactor, Set<File> files, String groupId,
428                               String artifactId, String oldVersion )
429     {
430 
431         getLog().debug( "Applying change " + groupId + ":" + artifactId + ":" + oldVersion + " -> " + newVersion );
432         // this is a triggering change
433         addChange( groupId, artifactId, oldVersion, newVersion );
434         // now fake out the triggering change
435 
436         Map.Entry<String, Model> current = PomHelper.getModelEntry( reactor, groupId, artifactId );
437         if ( current != null )
438         {
439             current.getValue().setVersion( newVersion );
440             addFile( files, project, current.getKey() );
441         }
442 
443         for ( Map.Entry<String, Model> sourceEntry : reactor.entrySet() )
444         {
445             final String sourcePath = sourceEntry.getKey();
446             final Model sourceModel = sourceEntry.getValue();
447 
448             getLog().debug( sourcePath.length() == 0 ? "Processing root module as parent"
449                                 : "Processing " + sourcePath + " as a parent." );
450 
451             final String sourceGroupId = PomHelper.getGroupId( sourceModel );
452             if ( sourceGroupId == null )
453             {
454                 getLog().warn( "Module " + sourcePath + " is missing a groupId." );
455                 continue;
456             }
457             final String sourceArtifactId = PomHelper.getArtifactId( sourceModel );
458             if ( sourceArtifactId == null )
459             {
460                 getLog().warn( "Module " + sourcePath + " is missing an artifactId." );
461                 continue;
462             }
463             final String sourceVersion = PomHelper.getVersion( sourceModel );
464             if ( sourceVersion == null )
465             {
466                 getLog().warn( "Module " + sourcePath + " is missing a version." );
467                 continue;
468             }
469 
470             addFile( files, project, sourcePath );
471 
472             getLog().debug(
473                 "Looking for modules which use " + ArtifactUtils.versionlessKey( sourceGroupId, sourceArtifactId )
474                     + " as their parent" );
475 
476             for ( Map.Entry<String, Model> stringModelEntry : processAllModules ? reactor.entrySet()
477                 : PomHelper.getChildModels( reactor, sourceGroupId, sourceArtifactId ).entrySet() )
478             {
479                 final Model targetModel = stringModelEntry.getValue();
480                 final Parent parent = targetModel.getParent();
481                 getLog().debug( "Module: " + stringModelEntry.getKey() );
482                 if ( parent != null && sourceVersion.equals( parent.getVersion() ) )
483                 {
484                     getLog().debug(
485                         "    parent already is " + ArtifactUtils.versionlessKey( sourceGroupId, sourceArtifactId )
486                             + ":" + sourceVersion );
487                 }
488                 else
489                 {
490                     getLog().debug(
491                         "    parent is " + ArtifactUtils.versionlessKey( sourceGroupId, sourceArtifactId ) + ":" + (
492                             parent == null ? "" : parent.getVersion() ) );
493                     getLog().debug(
494                         "    will become " + ArtifactUtils.versionlessKey( sourceGroupId, sourceArtifactId ) + ":"
495                             + sourceVersion );
496                 }
497                 final boolean targetExplicit = PomHelper.isExplicitVersion( targetModel );
498                 if ( ( updateMatchingVersions || !targetExplicit ) //
499                     && ( parent != null && StringUtils.equals( parent.getVersion(),
500                                                                PomHelper.getVersion( targetModel ) ) ) )
501                 {
502                     getLog().debug(
503                         "    module is " + ArtifactUtils.versionlessKey( PomHelper.getGroupId( targetModel ),
504                                                                          PomHelper.getArtifactId( targetModel ) ) + ":"
505                             + PomHelper.getVersion(
506                             targetModel ) );
507                     getLog().debug(
508                         "    will become " + ArtifactUtils.versionlessKey( PomHelper.getGroupId( targetModel ),
509                                                                            PomHelper.getArtifactId( targetModel ) )
510                             + ":" + sourceVersion );
511                     addChange( PomHelper.getGroupId( targetModel ), PomHelper.getArtifactId( targetModel ),
512                                PomHelper.getVersion( targetModel ), sourceVersion );
513                     targetModel.setVersion( sourceVersion );
514                 }
515                 else
516                 {
517                     getLog().debug(
518                         "    module is " + ArtifactUtils.versionlessKey( PomHelper.getGroupId( targetModel ),
519                                                                          PomHelper.getArtifactId( targetModel ) ) + ":"
520                             + PomHelper.getVersion(
521                             targetModel ) );
522                 }
523             }
524         }
525     }

527     private void addFile( Set<File> files, MavenProject project, String relativePath )
528     {
529         final File moduleDir = new File( project.getBasedir(), relativePath );
530         final File projectBaseDir = project.getBasedir();
531 
532         final File moduleProjectFile;
533 
534         if ( projectBaseDir.equals( moduleDir ) )
535         {
536             moduleProjectFile = project.getFile();
537         }
538         else if ( moduleDir.isDirectory() )
539         {
540             moduleProjectFile = new File( moduleDir, "pom.xml" );
541         }
542         else
543         {
544             // i don't think this should ever happen... but just in case
545             // the module references the file-name
546             moduleProjectFile = moduleDir;
547         }
548 
549         files.add( moduleProjectFile );
550     }

552     /**
553      * Updates the pom file.
554      *
555      * @param pom The pom file to update.
556      * @throws org.apache.maven.plugin.MojoExecutionException when things go wrong.
557      * @throws org.apache.maven.plugin.MojoFailureException   when things go wrong.
558      * @throws javax.xml.stream.XMLStreamException            when things go wrong.
559      */
560     protected synchronized void update( ModifiedPomXMLEventReader pom )
561         throws MojoExecutionException, MojoFailureException, XMLStreamException
562     {
563         ContextualLog log = new DelegatingContextualLog( getLog() );
564         try
565         {
566             Model model = PomHelper.getRawModel( pom );
567             log.setContext( "Processing " + PomHelper.getGroupId( model ) + ":" + PomHelper.getArtifactId( model ) );
568 
569             VersionChangerFactory versionChangerFactory = new VersionChangerFactory();
570             versionChangerFactory.setPom( pom );
571             versionChangerFactory.setLog( log );
572             versionChangerFactory.setModel( model );
573 
574             VersionChanger changer =
575                 versionChangerFactory.newVersionChanger( processParent, processProject, processDependencies,
576                                                          processPlugins );
577 
578             for ( VersionChange versionChange : sourceChanges )
579             {
580                 changer.apply( versionChange );
581             }
582 
583             if ( updateBuildOutputTimestamp && !"never".equals( updateBuildOutputTimestampPolicy ) )
584             {
585                 if ( "always".equals( updateBuildOutputTimestampPolicy ) || !sourceChanges.isEmpty() )
586                 {
587                     // also update project.build.outputTimestamp
588                     updateBuildOutputTimestamp( pom, model );
589                 }
590             }
591         }
592         catch ( IOException e )
593         {
594             throw new MojoExecutionException( e.getMessage(), e );
595         }
596         log.clearContext();
597     }

599     private void updateBuildOutputTimestamp( ModifiedPomXMLEventReader pom, Model model ) throws XMLStreamException
600     {
601         String buildOutputTimestamp = model.getProperties().getProperty( "project.build.outputTimestamp" );
602 
603         if ( buildOutputTimestamp == null || StringUtils.isEmpty( buildOutputTimestamp ) )
604         {
605             // no Reproducible Builds output timestamp defined
606             return;
607         }
608 
609         if ( StringUtils.isNumeric( buildOutputTimestamp ) )
610         {
611             // int representing seconds since the epoch, like SOURCE_DATE_EPOCH
612             buildOutputTimestamp = String.valueOf( System.currentTimeMillis() / 1000 );
613         }
614         else if ( buildOutputTimestamp.length() <= 1 )
615         {
616             // value length == 1 means disable Reproducible Builds
617             return;
618         }
619         else
620         {
621             // ISO-8601
622             DateFormat df = new SimpleDateFormat( "yyyy-MM-dd'T'HH:mm:ss'Z'" );
623             df.setTimeZone( TimeZone.getTimeZone( "UTC" ) );
624             buildOutputTimestamp = df.format( new Date() );
625         }
626 
627         PomHelper.setPropertyVersion( pom, null, "project.build.outputTimestamp", buildOutputTimestamp );
628     }
