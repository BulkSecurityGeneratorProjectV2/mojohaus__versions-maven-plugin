1   package org.codehaus.mojo.versions.filtering;
2   
3   import java.util.Collections;
4   import java.util.EnumMap;
5   import java.util.Map;
6   import java.util.function.Function;
7   import java.util.function.Predicate;
8   
9   import org.apache.maven.model.Dependency;
10  
11  public class TokenizedMatcher implements Predicate<Dependency>
12  {
13      public enum Tokens
14      {
15          GROUP_ID( Dependency::getGroupId ),
16          ARTIFACT_ID( Dependency::getArtifactId ),
17          VERSION( Dependency::getVersion ),
18          TYPE( Dependency::getType ),
19          CLASSIFIER( Dependency::getClassifier ),
20          SCOPE( Dependency::getScope );
21  
22          private final Function<Dependency, String> tokenExtractor;
23  
24          Tokens( Function<Dependency, String> tokenExtractor )
25          {
26              this.tokenExtractor = tokenExtractor;
27          }
28  
29          public Function<Dependency, String> getTokenExtractor()
30          {
31              return tokenExtractor;
32          }
33      }
34  
35      private final Map<Tokens, Predicate<String>> matchers;
36  
37      private TokenizedMatcher( Map<Tokens, Predicate<String>> matchers )
38      {
39          this.matchers = matchers;
40      }
41  
42      @Override
43      public boolean test( Dependency dependency )
44      {
45          for ( Tokens token : Tokens.values() )
46          {
47              String tokenValue = token.getTokenExtractor().apply( dependency );
48  
49              Predicate<String> matcher = matchers.get( token );
50              boolean matches = matcher.test( tokenValue );
51  
52              if ( !matches )
53              {
54                  return false;
55              }
56          }
57  
58          return true;
59      }
60  
61      public static TokenizedMatcher parse( String pattern )
62      {
63          EnumMap<Tokens, Predicate<String>> matchers = new EnumMap<>( Tokens.class );
64  
65          String[] split = pattern == null
66                  ? new String[0]
67                  : pattern.split( ":" );
68  
69          String groupIdPattern = split.length >= 1
70                  ? split[0]
71                  : WildcardMatcher.WILDCARD;
72          matchers.put( Tokens.GROUP_ID, WildcardMatcher.parse( groupIdPattern ) );
73  
74          String artifactIdPattern = split.length >= 2
75                  ? split[1]
76                  : WildcardMatcher.WILDCARD;
77          matchers.put( Tokens.ARTIFACT_ID, WildcardMatcher.parse( artifactIdPattern ) );
78  
79          String versionPattern = split.length >= 3
80                  ? split[2]
81                  : WildcardMatcher.WILDCARD;
82          matchers.put( Tokens.VERSION, WildcardMatcher.parse( versionPattern ) );
83  
84          String typePattern = split.length >= 4
85                  ? split[3]
86                  : WildcardMatcher.WILDCARD;
87          matchers.put( Tokens.TYPE, WildcardMatcher.parse( typePattern ) );
88  
89          String classifierPattern = split.length >= 5
90                  ? split[4]
91                  : WildcardMatcher.WILDCARD;
92          matchers.put( Tokens.CLASSIFIER, new NullAwareWildcardMatcher( classifierPattern ) );
93  
94          String scopePattern = split.length >= 6
95                  ? split[5]
96                  : WildcardMatcher.WILDCARD;
97          matchers.put( Tokens.SCOPE, new NullAwareWildcardMatcher( scopePattern ) );
98  
99          return new TokenizedMatcher( Collections.unmodifiableMap( matchers ) );
100     }
101 
102     @Override
103     public String toString()
104     {
105         return String.format( "%s{%s}", getClass().getSimpleName(), matchers );
106     }
107 }
