36  /**
37   * XML renderer for DependencyUpdatesReport creates an xml file in target directory and writes report about available
38   * dependency/dependency management updates.
39   *
40   * @author Illia Dubinin
41   * @since 2.4
42   */
43  public class DependencyUpdatesXmlRenderer
44  {
45  
46      private static final String GROUP_ID = "groupId";
47  
48      private static final String ARTIFACT_ID = "artifactId";
49  
50      private static final String SCOPE = "scope";
51  
52      private static final String CLASSIFIER = "classifier";
53  
54      private static final String TYPE = "type";
55  
56      private static final String CURRENT_VERSION = "currentVersion";
57  
58      private static final String NEXT_VERSION = "nextVersion";
59  
60      private static final String STATUS = "status";
61  
62      private static final String OPEN_TAG = "<";
63  
64      private static final String CLOSE_TAG = ">";
65  
66      private static final String OPEN_CLOSING_TAG = "</";
67  
68      private static final String NL = "\n";
69  
70      private static final String TAB = "\t";
71  
72      private Map<Dependency, ArtifactVersions> dependencyUpdates;
73  
74      private Map<Dependency, ArtifactVersions> dependencyManagementUpdates;
75  
76      private String outputFileName;
77  
78      public DependencyUpdatesXmlRenderer( Map<Dependency, ArtifactVersions> dependencyUpdates,
79                                           Map<Dependency, ArtifactVersions> dependencyManagementUpdates,
80                                           String outputFileName )
81      {
82          this.dependencyUpdates = dependencyUpdates;
83          this.dependencyManagementUpdates = dependencyManagementUpdates;
84          this.outputFileName = outputFileName;
85      }
86  
87      /**
88       * Makes report file with given name in target directory.
89       *
90       * @throws MavenReportException if something went wrong
91       */
92      public void render()
93          throws MavenReportException
94      {
95          StringBuilder sb = new StringBuilder();
96          sb.append( "<DependencyUpdatesReport>" ).append( NL );
97          Map<Dependency, ArtifactVersions> allUpdates = new TreeMap<>( new DependencyComparator() );
98          allUpdates.putAll( dependencyManagementUpdates );
99          allUpdates.putAll( dependencyUpdates );
100         sb.append( getSummaryBlock( allUpdates.values() ) );
101         sb.append( getDependencyInfoBlock( dependencyManagementUpdates, "dependencyManagements",
102                                            "dependencyManagement" ) );
103         sb.append( getDependencyInfoBlock( dependencyUpdates, "dependencies", "dependency" ) );
104         sb.append( "</DependencyUpdatesReport>" ).append( NL );
105         PrintWriter pw;
106         try
107         {
108             pw = new PrintWriter( outputFileName, "UTF8" );
109             pw.print( sb );
110             pw.close();
111         }
112         catch ( IOException e )
113         {
114             throw new MavenReportException( "Cannot create xml report.", e );
115         }
116     }

118     /**
119      * Method wraps value in xml tag. In ex: to wrap foo in tag bar you have to pass foo as value and bar as tag. As a
120      * result you will get: &lt;bar&gt;foo&lt;/bar&gt;
121      *
122      * @param value - string to wrap
123      * @param tag   - name of tag
124      * @return value wrapped in xml tag
125      */
126 
127     public static String wrapElement( String value, String tag )
128     {
129         return OPEN_TAG + tag + CLOSE_TAG + value + OPEN_CLOSING_TAG + tag + CLOSE_TAG;
130     }

132     /**
133      * Returns summary of dependency analysis result in xml format: current version, next available, next incremental,
134      * next minor and next major versions.
135      *
136      * @param allUpdates all dependencies versions
137      * @return summary in xml format
138      */
139     public static String getSummaryBlock( Collection<ArtifactVersions> allUpdates )
140     {
141         int numInc = 0;
142         int numMin = 0;
143         int numMaj = 0;
144         int numAny = 0;
145         int numCur = 0;
146         for ( ArtifactVersions details : allUpdates )
147         {
148             if ( details.getOldestUpdate( UpdateScope.SUBINCREMENTAL ) != null )
149             {
150                 numAny++;
151             }
152             else if ( details.getOldestUpdate( UpdateScope.INCREMENTAL ) != null )
153             {
154                 numInc++;
155             }
156             else if ( details.getOldestUpdate( UpdateScope.MINOR ) != null )
157             {
158                 numMin++;
159             }
160             else if ( details.getOldestUpdate( UpdateScope.MAJOR ) != null )
161             {
162                 numMaj++;
163             }
164             else
165             {
166                 numCur++;
167             }
168         }
169 
170         String result = "\t<summary>%n" + "\t\t<usingLastVersion>%d</usingLastVersion>%n"
171             + "\t\t<nextVersionAvailable>%d</nextVersionAvailable>%n"
172             + "\t\t<nextIncrementalAvailable>%d</nextIncrementalAvailable>%n"
173             + "\t\t<nextMinorAvailable>%d</nextMinorAvailable>%n" + "\t\t<nextMajorAvailable>%d</nextMajorAvailable>%n"
174             + "\t</summary>%n";
175 
176         return String.format( result, numCur, numAny, numInc, numMin, numMaj );
177     }

179     /**
180      * Returns xml report for current dependency state with following info: current version, next available version,
181      * next incremental/minor/major if available and status ('incremental available', 'minor available', 'major
182      * available' or 'no new available')
183      *
184      * @param versions version info for dependency
185      * @return xml reports about current possible updates.
186      */
187     public static String getVersionsBlocks( ArtifactVersions versions )
188     {
189         StringBuilder sBuilder = new StringBuilder();
190         sBuilder.append( TAB ).append( TAB ).append( TAB ).append( wrapElement( versions.isCurrentVersionDefined()
191                         ? versions.getCurrentVersion().toString() : versions.getArtifact().getVersionRange().toString(),
192                 CURRENT_VERSION ) ).append( NL );
193         ArtifactVersion nextVersion = versions.getOldestUpdate( UpdateScope.ANY );
194         if ( nextVersion != null )
195         {
196             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( wrapElement( nextVersion.toString(),
197                                                                                     NEXT_VERSION ) ).append( NL );
198 
199             String incrementalsBlock = getVersionsInScopeBlock( versions, UpdateScope.INCREMENTAL );
200             sBuilder.append( incrementalsBlock );
201             String minorsBlock = getVersionsInScopeBlock( versions, UpdateScope.MINOR );
202             sBuilder.append( minorsBlock );
203             String majorsBlock = getVersionsInScopeBlock( versions, UpdateScope.MAJOR );
204             sBuilder.append( majorsBlock );
205 
206             String status = null;
207             if ( incrementalsBlock.length() > 0 )
208             {
209                 status = "incremental available";
210             }
211             else if ( minorsBlock.length() > 0 )
212             {
213                 status = "minor available";
214             }
215             else if ( majorsBlock.length() > 0 )
216             {
217                 status = "major available";
218             }
219 
220             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( wrapElement( status, STATUS ) ).append( NL );
221         }
222         else
223         {
224             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( wrapElement( "no new available",
225                                                                                     STATUS ) ).append( NL );
226         }
227 
228         return sBuilder.toString();
229     }

231     private static String getDependencyInfoBlock( Map<Dependency, ArtifactVersions> dependencyUpdates, String blockName,
232                                                   String subblockName )
233     {
234         StringBuilder sBuilder = new StringBuilder();
235         sBuilder.append( TAB ).append( OPEN_TAG ).append( blockName ).append( CLOSE_TAG ).append( NL );
236         for ( Entry<Dependency, ArtifactVersions> entry : dependencyUpdates.entrySet() )
237         {
238             sBuilder.append( TAB ).append( TAB ).append( OPEN_TAG ).append( subblockName ).append( CLOSE_TAG )
239                 .append( NL );
240 
241             Dependency dep = entry.getKey();
242             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( wrapElement( dep.getGroupId(),
243                                                                                     GROUP_ID ) ).append( NL );
244             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( wrapElement( dep.getArtifactId(),
245                                                                                     ARTIFACT_ID ) ).append( NL );
246             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( wrapElement( dep.getScope(),
247                                                                                     SCOPE ) ).append( NL );
248             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( wrapElement( dep.getClassifier(),
249                                                                                     CLASSIFIER ) ).append( NL );
250             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( wrapElement( dep.getType(),
251                                                                                     TYPE ) ).append( NL );
252 
253             sBuilder.append( getVersionsBlocks( entry.getValue() ) );
254 
255             sBuilder.append( TAB ).append( TAB ).append( OPEN_CLOSING_TAG ).append( subblockName ).append( CLOSE_TAG )
256                 .append( NL );
257         }
258         sBuilder.append( TAB ).append( OPEN_CLOSING_TAG ).append( blockName ).append( CLOSE_TAG ).append( NL );
259         return sBuilder.toString();
260     }
261 
262     private static String getVersionsInScopeBlock( ArtifactVersions av, UpdateScope scope )
263     {
264         String versionsTag = scope.toString().toLowerCase() + "s";
265         StringBuilder sBuilder = new StringBuilder();
266 
267         ArtifactVersion nextVersion = av.getOldestUpdate( scope );
268         if ( nextVersion != null )
269         {
270             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( OPEN_TAG ).append( versionsTag )
271                 .append( CLOSE_TAG ).append( NL );
272             ArtifactVersion[] versions = av.getAllUpdates( scope );
273             for ( ArtifactVersion version : versions )
274             {
275                 sBuilder.append( TAB ).append( TAB ).append( TAB ).append( TAB )
276                     .append( wrapElement( version.toString(),
277                                           scope.toString().toLowerCase() ) ).append( NL );
278             }
279             sBuilder.append( TAB ).append( TAB ).append( TAB ).append( OPEN_CLOSING_TAG ).append( versionsTag )
280                 .append( CLOSE_TAG ).append( NL );
281         }
282 
283         return sBuilder.toString();
284     }
