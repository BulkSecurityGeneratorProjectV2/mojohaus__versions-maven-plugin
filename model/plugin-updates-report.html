PluginUpdatesReport

XML version of the Plugin Updates Report

<PluginUpdatesReport xmlns="https://www.mojohaus.org/VERSIONS/PLUGIN-UPDATES-REPORT/2.1.0" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
  xsi:schemaLocation="https://www.mojohaus.org/VERSIONS/PLUGIN-UPDATES-REPORT/2.1.0 https://www.mojohaus.org/versions-maven-plugin/xsd/plugin-updates-report-2.1.0.xsd">
  <summary>
    <usingLastVersion/>
    <nextVersionAvailable/>
    <nextIncrementalAvailable/>
    <nextMinorAvailable/>
    <nextMajorAvailable/>
    <dependencyUpdates/>
  </summary>
  <pluginManagements>
    <pluginManagement>
      <groupId/>
      <artifactId/>
      <scope/>
      <classifier/>
      <type/>
      <currentVersion/>
      <lastVersion/>
      <any/>
      <incrementals/>
      <minors/>
      <majors/>
      <status/>
    </pluginManagement>
  </pluginManagements>
  <plugins>
    <plugin>
      <groupId/>
      <artifactId/>
      <scope/>
      <classifier/>
      <type/>
      <currentVersion/>
      <lastVersion/>
      <any/>
      <incrementals/>
      <minors/>
      <majors/>
      <status/>
    </plugin>
  </plugins>
</PluginUpdatesReport>
