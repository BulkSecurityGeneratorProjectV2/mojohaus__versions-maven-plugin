117     @Test
118     public void testRemoveSnapshotIdempotency()
119             throws Exception
120     {
121         Path pomDir = Files.createTempDirectory( "set-" );
122         try
123         {
124             Files.copy( Paths.get( "src/test/resources/org/codehaus/mojo/set/remove-snapshot/pom.xml" ),
125                     Paths.get( pomDir.toString(),  "pom.xml" ), REPLACE_EXISTING );
126 
127             SetMojo firstRun = (SetMojo) mojoRule.lookupConfiguredMojo( pomDir.toFile(), "set" );
128             firstRun.execute();
129             assertThat( String.join( "", Files.readAllLines( Paths.get( pomDir.toString(), "pom.xml" ) ) ),
130                     containsString( "<version>1.0</version>" ) );
131 
132             // no exception should be thrown, the file should stay with version "1.0"
133             SetMojo secondRun = (SetMojo) mojoRule.lookupConfiguredMojo( pomDir.toFile(), "set" );
134             MavenExecutionRequest request =
135                     (MavenExecutionRequest) getVariableValueFromObject( secondRun.settings, "request" );
136             setVariableValueToObject( request, "interactiveMode", false );
137             secondRun.execute();
138             assertThat( String.join( "", Files.readAllLines( Paths.get( pomDir.toString(), "pom.xml" ) ) ),
139                     containsString( "<version>1.0</version>" ) );
140         }
141         finally
142         {
143             if ( pomDir != null && pomDir.toFile().exists() )
144             {
145                 Arrays.stream( Objects.requireNonNull( pomDir.toFile().listFiles() ) ).forEach( File::delete );
146                 pomDir.toFile().delete();
147             }
148         }
