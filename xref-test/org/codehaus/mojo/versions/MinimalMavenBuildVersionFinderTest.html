3   import org.apache.maven.artifact.versioning.DefaultArtifactVersion;
4   import org.junit.Test;
5   
6   import static org.junit.Assert.assertEquals;
7   import static org.junit.Assert.assertNull;

9   /**
10   * Unit tests for MinimalMavenBuildVersionFinder.
11   */
12  public class MinimalMavenBuildVersionFinderTest
13  {
14  
15      @Test
16      public void testValidVersionRanges()
17      {
18          DefaultArtifactVersion expectedMinimumVersion = new DefaultArtifactVersion( "1.0" );
19          assertEquals( expectedMinimumVersion, MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "1.0" ) );
20          assertEquals( expectedMinimumVersion, MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "[1.0]" ) );
21          assertEquals( expectedMinimumVersion, MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "[1.0,)" ) );
22          assertEquals( expectedMinimumVersion, MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "(1.0,)" ) );
23          assertEquals( expectedMinimumVersion,
24                        MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "[1.0,2.0]" ) );
25          assertEquals( expectedMinimumVersion,
26                        MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "(1.0,2.0)" ) );
27          assertEquals( expectedMinimumVersion,
28                        MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "(1.0,2.0]" ) );
29          assertEquals( expectedMinimumVersion,
30                        MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "[1.0,2.0)" ) );
31      }
32  
33      @Test
34      public void testInvalidVersionRanges()
35      {
36          assertNull( MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "(,1.0]" ) );
37          assertNull( MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "(,1.0)" ) );
38          assertNull( MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "(,1.0],[1.2,)" ) );
39          assertNull( MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "(,1.1),(1.1,)" ) );
40          assertNull( MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "(,1.1),[1.1,)" ) );
41          assertNull( MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "(1.0" ) );
42          assertNull( MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "1.0)" ) );
43          assertNull( MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "1.0()" ) );
44          assertNull( MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "()1.0" ) );
45          assertNull( MinimalMavenBuildVersionFinder.getMinimumVersionFromRange( "(1.0]" ) );
46      }
