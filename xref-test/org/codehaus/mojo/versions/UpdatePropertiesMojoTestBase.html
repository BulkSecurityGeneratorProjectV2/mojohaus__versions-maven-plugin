21  import java.nio.file.Path;
22  import java.util.HashMap;
23  
24  import org.apache.maven.artifact.metadata.ArtifactMetadataSource;
25  import org.apache.maven.plugin.Mojo;
26  import org.apache.maven.plugin.testing.AbstractMojoTestCase;
27  import org.apache.maven.plugin.testing.MojoRule;
28  import org.apache.maven.plugin.testing.stubs.StubArtifactRepository;
29  import org.codehaus.mojo.versions.utils.TestChangeRecorder;
30  import org.junit.After;
31  import org.junit.Before;
32  import org.junit.Rule;
33  
34  import static org.codehaus.mojo.versions.utils.MockUtils.mockArtifactMetadataSource;
35  import static org.codehaus.mojo.versions.utils.TestUtils.createTempDir;
36  import static org.codehaus.mojo.versions.utils.TestUtils.tearDownTempDir;
37  
38  /**
39   * Base class for {@link UpdatePropertiesMojo} and {@link UpdatePropertyMojo} test suites
40   */
41  public abstract class UpdatePropertiesMojoTestBase extends AbstractMojoTestCase
42  {
43      @Rule
44      public MojoRule mojoRule = new MojoRule( this );
45      protected Path pomDir;
46      protected ArtifactMetadataSource artifactMetadataSource;
47      protected TestChangeRecorder changeRecorder;
48  
49      @Before
50      public void setUp() throws Exception
51      {
52          super.setUp();
53          pomDir = createTempDir( "update-property" );
54          changeRecorder = new TestChangeRecorder();
55          artifactMetadataSource = mockArtifactMetadataSource( new HashMap<String, String[]>()
56          {{
57              put( "default-artifact", new String[] {"1.0.0", "1.0.1-rc1", "1.1.0-alpha", "2.0.0-M1"} );
58          }} );
59      }
60  
61      @After
62      public void tearDown() throws Exception
63      {
64          try
65          {
66              tearDownTempDir( pomDir );
67          }
68          finally
69          {
70              super.tearDown();
71          }
72      }
73  
74      @SuppressWarnings( "unchecked" )
75      protected  <T extends Mojo> T setUpMojo( String goal ) throws Exception
76      {
77          T mojo = (T) mojoRule.lookupConfiguredMojo( pomDir.toFile(), goal );
78          setVariableValueToObject( mojo, "localRepository", new StubArtifactRepository( pomDir.toString() ) );
79          setVariableValueToObject( mojo, "artifactMetadataSource", artifactMetadataSource );
80          setVariableValueToObject( mojo, "changeRecorder", changeRecorder );
81          setVariableValueToObject( mojo, "generateBackupPoms", false );
82          return (T) mojo;
83      }
84  }
