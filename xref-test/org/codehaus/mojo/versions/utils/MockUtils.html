44  /**
45   * Various mock creating utilities
46   */
47  public class MockUtils
48  {
49      private static final Map<String, String[]> DEFAULT_VERSION_MAP = new HashMap<String, String[]>()
50      {{
51          put( "artifactA", new String[] {"1.0.0", "2.0.0"} );
52          put( "artifactB", new String[] {"1.0.0", "1.1.0"} );
53          put( "artifactC", new String[] {"1.0.0"} );
54      }};
55  
56      /**
57       * Creates a mocked  {@linkplain ArtifactMetadataSource}, providing the default version set
58       * @return mocked {@linkplain ArtifactMetadataSource}
59       */
60      public static ArtifactMetadataSource mockArtifactMetadataSource()
61      {
62          return mockArtifactMetadataSource( DEFAULT_VERSION_MAP );
63      }
64  
65      /**
66       * Creates a mocked  {@linkplain ArtifactMetadataSource}, providing the version map given in the argument
67       * @param versionMap requested version map
68       * @return mocked {@linkplain ArtifactMetadataSource}
69       */
70      public static ArtifactMetadataSource mockArtifactMetadataSource( Map<String, String[]> versionMap )
71      {
72          ArtifactMetadataSource artifactMetadataSource = mock( ArtifactMetadataSource.class );
73          try
74          {
75              when( artifactMetadataSource.retrieveAvailableVersions( any( Artifact.class ), any(), any() ) ).then(
76                      invocation ->
77                      {
78                          Artifact artifact = invocation.getArgument( 0 );
79                          return versionMap.entrySet().stream()
80                                  .filter( e -> e.getKey().equals( artifact.getArtifactId() ) )
81                                  .findAny()
82                                  .map( e -> Arrays.stream( e.getValue() )
83                                          .map( DefaultArtifactVersion::new )
84                                          .collect( ArrayList::new, ArrayList::add, ArrayList::add ) )
85                                  .orElse( null ); // should tell us if we haven't populated all cases in the test
86                      } );
87          }
88          catch ( ArtifactMetadataRetrievalException e )
89          {
90              throw new RuntimeException( e );
91          }
92          return artifactMetadataSource;
93      }
94  
95      public static I18N mockI18N()
96      {
97          I18N i18n = mock( I18N.class );
98          when( i18n.getString( anyString(), any(), anyString() ) ).thenAnswer(
99              invocation -> invocation.getArgument( 2 ) );
100         return i18n;
101     }
102 
103     public static SiteTool mockSiteTool()
104     {
105         Artifact skinArtifact = mock( Artifact.class );
106         when( skinArtifact.getId() ).thenReturn( "" );
107         SiteTool siteTool = mock( SiteTool.class );
108         try
109         {
110             when( siteTool.getSkinArtifactFromRepository( any(), any(), any() ) ).thenReturn( skinArtifact );
111         }
112         catch ( SiteToolException e )
113         {
114             throw new RuntimeException( e );
115         }
116         return siteTool;
117     }
118 
119     public static RepositorySystem mockRepositorySystem()
120     {
121         RepositorySystem repositorySystem = mock( RepositorySystem.class );
122         when( repositorySystem.createDependencyArtifact( any( Dependency.class ) ) ).thenAnswer(
123             invocation ->
124             {
125                 Dependency dependency = invocation.getArgument( 0 );
126                 return new DefaultArtifact( dependency.getGroupId(), dependency.getArtifactId(),
127                                             dependency.getVersion(), dependency.getScope(), dependency.getType(),
128                                             dependency.getClassifier(), new DefaultArtifactHandlerStub( "default" ) );
129             } );
130         return repositorySystem;
131     }
