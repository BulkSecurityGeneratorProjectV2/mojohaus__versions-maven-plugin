1   package org.codehaus.mojo.versions;
2   
3   import java.util.Objects;
4   
5   import org.apache.maven.model.Dependency;
6   import org.hamcrest.Description;
7   import org.hamcrest.TypeSafeMatcher;
8   
9   /**
10   * Hamcrest-Matcher that matches a {@link Dependency} GAV
11   */
12  public class HasGAVMatcher extends TypeSafeMatcher<Dependency>
13  {
14      private final String groupId;
15      private final String artifactId;
16      private final String version;
17  
18      public HasGAVMatcher( String groupId, String artifactId, String version )
19      {
20          this.groupId = groupId;
21          this.artifactId = artifactId;
22          this.version = version;
23      }
24  
25      public static HasGAVMatcher hasGAVOf( Dependency dependency )
26      {
27          return hasGAV( dependency.getGroupId(), dependency.getArtifactId(), dependency.getVersion() );
28      }
29  
30      public static HasGAVMatcher hasGAV( String groupId, String artifactId, String version )
31      {
32          return new HasGAVMatcher( groupId, artifactId, version );
33      }
34  
35      @Override
36      protected boolean matchesSafely( Dependency item )
37      {
38          boolean result = Objects.equals( groupId, item.getGroupId() )
39                  && Objects.equals( artifactId, item.getArtifactId() )
40                  && Objects.equals( version, item.getVersion() );
41  
42          return result;
43      }
44  
45      @Override
46      public void describeTo( Description description )
47      {
48          description.appendText( String.format( "has GAV %s:%s:%s", groupId, artifactId, version ) );
49      }
50  }
